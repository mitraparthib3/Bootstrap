    {x: '2012 Q4', y: 7},
    {x: '2013 Q1', y: 4}
  ],
  xkey: 'x',
  ykeys: ['y'],
  labels: ['Y'],
  barColors: function (row, series, type) {
    if (type === 'bar') {
      var red = Math.ceil(255 * row.y / this.ymax);
      return 'rgb(' + red + ',250,235)';
    }
    else {
      return '#000';
    }
  }
});
